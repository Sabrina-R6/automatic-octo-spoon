import pandas as pd
import matplotlib.pyplot as plt
from collections import Counter

# Load the Excel file
file_path = r'C:\Users\SabrinaRummani\Downloads\Non Renewals - MLS-2025-05-07-07-39-08.xlsx'
df = pd.read_excel(file_path)

# Process and count occurrences
processes = df['Process'].dropna().str.split('; ')
flattened_processes = [proc.strip() for sublist in processes for proc in sublist]
process_counts = Counter(flattened_processes)

# Create a DataFrame and sort
process_df = pd.DataFrame(process_counts.items(), columns=['Process', 'Count']).sort_values(by='Count', ascending=True)

# Plot all processes
plt.figure(figsize=(10, max(6, len(process_df) * 0.4)))  # Dynamic height
bars = plt.barh(process_df['Process'], process_df['Count'], color='skyblue')
plt.xlabel('Number of Non-Renewals')
plt.title('All Processes Among Non-Renewing Members')

# Add data labels
for bar in bars:
    width = bar.get_width()
    plt.text(width + 0.2, bar.get_y() + bar.get_height()/2,
             str(int(width)), va='center')

plt.tight_layout()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from collections import Counter

# Load the Excel file
file_path = r'C:\Users\SabrinaRummani\Downloads\Non Renewals - MLS-2025-05-07-07-39-08.xlsx'
df = pd.read_excel(file_path)

# Ensure 'Lapsed On' is in datetime format
df['Lapsed On'] = pd.to_datetime(df['Lapsed On'], errors='coerce')

# Filter for September 2024 lapses
september_lapses = df[df['Lapsed On'].dt.to_period('M') == pd.Period('2024-09')]

# Extract and count processes
september_processes = september_lapses['Process'].dropna().str.split('; ')
flattened_sept_processes = [proc.strip() for sublist in september_processes for proc in sublist]
sept_process_counts = Counter(flattened_sept_processes)

# Create and sort DataFrame
sept_process_df = pd.DataFrame(sept_process_counts.items(), columns=['Process', 'Count']).sort_values(by='Count', ascending=True)

# Create lollipop chart
plt.figure(figsize=(10, max(6, len(sept_process_df) * 0.4)))
y = sept_process_df['Process']
x = sept_process_df['Count']

plt.plot(x, y, "o", markersize=8, color='seagreen')

# Add labels
for i in range(len(x)):
    plt.text(x.iloc[i] + 0.2, y.iloc[i], str(x.iloc[i]), va='center')

plt.title('Lollipop Chart of Processes (September 2024 Non-Renewals)')
plt.xlabel('Number of Non-Renewals')
plt.tight_layout()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from collections import Counter
import matplotlib.ticker as ticker  # <-- for integer axis formatting

# Load the Excel file
file_path = r'C:\Users\SabrinaRummani\Downloads\Non Renewals - MLS-2025-05-07-07-39-08.xlsx'
df = pd.read_excel(file_path)

# Ensure 'Lapsed On' is in datetime format
df['Lapsed On'] = pd.to_datetime(df['Lapsed On'], errors='coerce')

# Filter for July 2024 lapses
july_lapses = df[df['Lapsed On'].dt.to_period('M') == pd.Period('2024-07')]

# Extract and count processes
july_processes = july_lapses['Process'].dropna().str.split('; ')
flattened_july_processes = [proc.strip() for sublist in july_processes for proc in sublist]
july_process_counts = Counter(flattened_july_processes)

# Create and sort DataFrame
july_process_df = pd.DataFrame(july_process_counts.items(), columns=['Process', 'Count']).sort_values(by='Count', ascending=True)

# Create lollipop chart
plt.figure(figsize=(10, max(6, len(july_process_df) * 0.4)))
y = july_process_df['Process']
x = july_process_df['Count']
plt.hlines(y=y, xmin=0, xmax=x, color='gray', alpha=0.7, linewidth=2)
plt.plot(x, y, "o", markersize=8, color='dodgerblue')

# Add integer labels
for i in range(len(x)):
    plt.text(x.iloc[i] + 0.2, y.iloc[i], str(int(x.iloc[i])), va='center')

# Force integer ticks on x-axis
plt.gca().xaxis.set_major_locator(ticker.MaxNLocator(integer=True))

plt.title('Lollipop Chart of Processes (July 2024 Non-Renewals)')
plt.xlabel('Number of Non-Renewals')
plt.tight_layout()
plt.show()
