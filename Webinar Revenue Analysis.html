def assign_topic(name):
    name = str(name).lower()
    for topic, keywords in topic_keywords.items():

            return topic.capitalize()
    return 'Other'

# Copy keyword-topic dictionary
topic_keywords = {
    'standards': ['standard', 'specification'],
    'market': ['market', 'trends', 'analysis', 'industry'],
    'technology': ['tech', 'ai', 'robotics', 'software'],
    'sustainability': ['sustainability', 'green', 'recycle'],
    'education': ['training', 'education', 'learn'],
    'government': ['legislation', 'policy', 'government'],
    'business': ['business', 'strategy', 'growth', 'planning', 'developing', 'leadership'],
    'manufacturing & materials': ['manufacturing', 'production', 'pultrusion', 'material', 'composites'],
}

# Flatten keyword list
keyword_topic_map = {}
for topic, keywords in topic_keywords.items():
    for kw in keywords:
        keyword_topic_map[kw] = topic

# Function to extract matched keywords
def find_keywords(text):
    text = str(text).lower()
    matches = [kw for kw in keyword_topic_map if kw in text]
    return matches if matches else ['None']

# Apply to DataFrame
df['Matched Keywords'] = df['Event Name'].apply(find_keywords)


exploded_df = df.explode('Matched Keywords')

# Filter out rows with no match
keyword_df = exploded_df[exploded_df['Matched Keywords'] != 'None']

# Group by keyword and summarize
keyword_summary = keyword_df.groupby('Matched Keywords').agg(
    Event_Count=('Event Name', 'count'),
    Total_Registrants=('Total Registrants', 'sum'),
    Avg_Registrants=('Total Registrants', 'mean')
).sort_values(by='Avg_Registrants', ascending=False)

# Make whole numbers
keyword_summary = keyword_summary.round(0).astype(int)

display(keyword_summary)
